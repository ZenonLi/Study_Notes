# 一、LED

```c
#include <stdio.h>
#include <stdlib.h>
#include <wiringPi.h>
#include <unistd.h>

char Ledr = 27;
char Ledg = 28;
char Ledb = 29;

int main(){
    //wiringPiSetupGpio() 更改不同引脚方式
    if (wiringPiSetup() < 0){
        perror("启动树莓派GPIO失败");
        exit(1);
    }
    pinMode(Ledr, OUTPUT);
    pinMode(Ledg, OUTPUT);
    pinMode(Ledb, OUTPUT);
    while(1){
        digitalWrite(Ledr, 1);
        sleep(5);
        digitalWrite(Ledr, 0);
        break;
    }
}
```
